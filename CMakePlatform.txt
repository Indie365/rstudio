if (NOT WIN32)

  # detect the operating system's name
  if(NOT OS_NAME)
    execute_process(
      COMMAND "bash" "-c" "source ${PROJECT_ROOT_DIR}/dependencies/common/rstudio-tools; echo \$(platform)"
      OUTPUT_VARIABLE OS_NAME
      OUTPUT_STRIP_TRAILING_WHITESPACE
      )
    set(OS_NAME "${OS_NAME}" CACHE STRING "Operating system name")
  endif()

  # detect the operating system version
  if(NOT OS_VERSION)
    execute_process(
      COMMAND "bash" "-c" "source ${PROJECT_ROOT_DIR}/dependencies/common/rstudio-tools; echo \$(os-version)"
      OUTPUT_VARIABLE OS_VERSION
      OUTPUT_STRIP_TRAILING_WHITESPACE)
    set(OS_VERSION "${OS_VERSION}" CACHE STRING "Operating system version")
  endif()

endif()
